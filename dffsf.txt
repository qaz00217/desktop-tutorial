동적포인터변수연속계산굿:
#include <stdio.h>
#include <stdlib.h>

int main() {
    int *p;           // 포인터 선언
    p = (int*)malloc(sizeof(int)); // 정수 1개 동적 할당
    *p = 10;          // 값 저장
    printf("%d\n", *p); // 10 출력

    free(p);          // 메모리 해제
    return 0;
}
동적배열:
int *arr;
int n = 5;

arr = (int*)malloc(n * sizeof(int)); // int 5개 동적 할당

for (int i = 0; i < n; i++) {
    arr[i] = i * 10;
}

free(arr); // 사용 후 반드시 해제
